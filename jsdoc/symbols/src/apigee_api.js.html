1 /**
  2   * This library depends on jQuery and the base64 jQuery plugin (http://plugins.jquery.com/project/base64)
  3   * Sample Usage:
  4   * Working with the Apigee API
  5   * - Setting up an Apigee account:
  6   * var apigee = new $.apigee_api('https://api.apigee.com');
  7   * apigee.init("earth2marsh","supersecret"); // pwd will be base64 encoded
  8   *
  9   * - Adding an application:
 10   * apigee.request('post','/apps/myappname.json');
 11   *
 12   * Working with a proxied API
 13   * - Set up the app object:
 14   * var myapp = $.apigee_api('endpoint');
 15   *
 16   * - Casting an app to a user:
 17   * myapp.init("myappuser","myappuserpwd"); // base64 encode pwd
 18   *
 19   * - Making a request:
 20   * function renderTimeline(data) {
 21   *   // do clever things with that data, assume for now that rendering is outside the scope of the library
 22   * }
 23   * myapp.request('get','/1/statuses/home_timeline.json',{},{callback:'renderTimeline'}); // callback to custom function
 24 */
 25 
 26 (function($) {
 27 /**
 28   * Creates API object
 29   * Optionally accepts:
 30   * - endpoint
 31   * - username
 32   * - password
 33 */
 34   function ApigeeApi(endpoint,username,password) {
 35     var theApi = this;
 36 /**
 37   * Default verb, response type, and endpoint; can be overridden by passing in settings to request
 38 */
 39     this.defaults = {
 40       verb:"get",
 41       type:"json",
 42       endpoint:endpoint || "https://api.apigee.com"
 43     };
 44 /**
 45   * Makes request to endpoint + request
 46   * Excepts:
 47   * - verb (if not provided, defaults to the object's default verb (typically "get")
 48   * - request (NOT optional; request will not be made if not provided)
 49   * - headers (optional object - these are passed in as request headers or query parameters, depending on verb)
 50   * - settings (optional object - this allows defaults to be overridden, callback function to be provided, etc.)
 51   * Sample Request:
 52   * - (assuming "thisApi" is the name of an instance of the ApigeeApi object, and also that there is a function called "renderTimeline" that expects the response data as a parameter)
 53   * - thisApi.request('get','/1/statuses/home_timeline.json',{sort:'ascending'},{callback:'renderTimeline'});
 54 */
 55     this.request = function(verb,request,headers,settings) {
 56       if (request) {
 57         var returnObject = {};
 58         var verb = verb || theApi.defaults.verb;
 59         var headers = headers || {};
 60         if (theApi.smartkey) headers = $.extend(true, headers, {"smartkey":theApi.smartkey});
 61         var settings = $.extend({}, theApi.defaults, settings);
 62         var fullRequest = request.split(".")
 63         var request = fullRequest[0];
 64         if (fullRequest.length > 1) settings.type = fullRequest[1];
 65         if (settings.type === 'jsonp') settings.type = 'json';
 66         $.ajax({
 67           url: settings.endpoint+request+'.'+settings.type.split(" ")[0],
 68           headers: headers,
 69           data: headers,
 70           type: settings.verb,
 71           dataType: settings.type,
 72           success: function(data,textStatus,jqXHR) {
 73             console.log(textStatus);
 74             returnObject.response_message = textStatus;
 75             returnObject.payload = data;
 76             returnObject.xhr = jqXHR;
 77             if (settings.callback) {
 78               var callbackFunction = new Function(settings.callback+'(\''+data+'\')');
 79               callbackFunction();
 80             }     
 81           },
 82           error: function(jqXHR, textStatus, errorThrown) {
 83             responseMessage = textStatus+" ("+errorThrown+")";
 84             console.log(responseMessage);
 85             returnObject.response_message = responseMessage;
 86             returnObject.xhr = jqXHR;
 87           }
 88         });
 89         return returnObject;
 90       }
 91     }
 92 /**
 93   * base64-encodes username and password and makes request for smartkey.  If response is successful, adds this to the ApigeeAPI object, which in turn adds it as a header to each subsequent request
 94 */
 95     this.init = function(username,password) {
 96       var username = username || "";
 97       var password = password || "";
 98       $.ajax({
 99         url: "./sample.json",
100         headers: {"auth":$.base64Encode(username+':'+password)},
101         type: "get",
102         dataType: "json",
103         success: function(data,textStatus,jqXHR) {
104           if (data.smartkey) theApi.smartkey = data.smartkey;
105         }
106       });
107     };
108     if (username && password) this.init(username,password);
109   }  
110     
111   $.apigee_api = function(endpoint,username,password) {
112     return new ApigeeApi(endpoint || false,username || false,password || false);
113   }
114   
115 })(jQuery);
