  5 /**
  6   * Accepts the endpoint (ex: "http://user.apigee.com/") and builds the API Caller object
  7 */
  8 function ComApigeeApiCaller(endPoint) {
  9   var theCall = this;
 10   /**
 11     * Takes an API Request (ex: "statuses/public_timeline.json") and an optional array of specified call parameters.
 12     * theCall's rawResponse is set to the raw JSON object response.
 13     * If a DOM container ID was provided, response is also sent through processCall.
 14   */
 15   this.callAPI = function (apiRequest,callParams) {
 16     if (!apiRequest) var apiRequest = "";
 17     if (!callParams) var callParams = {};
 18     if (!callParams.containerId) callParams.containerId = null;
 19     if (!callParams.callVerb) callParams.callVerb = "GET";
 20     if (!callParams.htmlConstruct) callParams.htmlConstruct = "ul";
 21     if (!callParams.includeElements) callParams.includeElements = {};
 22     this.doPrune = false;
 23     for (var key in callParams.includeElements) {
 24       if (callParams.includeElements.hasOwnProperty(key)) {
 25         this.doPrune = true;
 26         break;
 27       }
 28     }
 29     var callAuth = new ComApigeeAuthHandler();
 30     var urlParams = {
 31       "auth" : $.base64Encode(callAuth.userName + ":" + callAuth.userPass),
 32       "apigee_verb" : callParams.callVerb,
 33       "apigee_testbed" : window.location.hostname
 34     };
 35     if (callParams.extraParams) $.extend(true, urlParams, callParams.extraParams);
 36     $.ajax({
 37       url: endPoint+apiRequest,
 38       dataType: 'jsonp',
 39       data: urlParams,
 40       success: function (data) {
 41         theCall.rawResponse = data;
 42         theCall.partialResponse = (theCall.doPrune) ? theCall.pruneResponse(data) : data;
 43         if (callParams.containerId !== null) {
 44           theCall.processCall(theCall.partialResponse,callParams.containerId,callParams.htmlConstruct);
 45         }
 46       }
 47     });
 48   }
 49   /**
 50     * Filters response data based on submitted prototype object (callParams.includeElements)
 51   */
 52   this.pruneResponse = function (rawData) {
 53     var prunedData = {};
 54     var pruneLoop = function(rawObject,protoObject,prunedObject) {
 55       for (var rawKey in rawObject) {
 56         if (rawObject.hasOwnProperty(rawKey) && protoObject.hasOwnProperty(rawKey)) {
 57           var rawVal = rawObject[rawKey];
 58           var protoVal = protoObject[rawKey];
 59           if ((typeof(protoVal)==='object') && protoVal !== null) {
 60             prunedObject[rawKey] = {};
 61             pruneLoop(rawVal,protoVal,prunedObject[rawKey]);
 62           } else {
 63             prunedObject[rawKey] = rawVal;
 64           }
 65         }
 66       }
 67     }
 68     for (var key in rawData) {
 69       if (rawData.hasOwnProperty(key)) {
 70         prunedData[key] = {};
 71         pruneLoop(rawData[key],callParams.includeElements,prunedData[key]);
 72       }
 73     }
 74     return prunedData;  
 75   }
 76   /**
 77     * Takes the request data, an optional DOM ID of the target container, and an HTML construct (ul, ol, div, or dl).
 78     * setConstruct and iterateData format the response; the target container is populated accordingly.
 79   */
 80   this.processCall = function (data,containerId,htmlConstruct) {
 81     theCall.setConstruct(htmlConstruct);
 82     var outputHTML = this.iterateData(data);
 83     $('#'+containerId).html(outputHTML);
 84   }
 85   /**
 86     * Loops through the response data and formats it based on the construct.
 87   */
 88   this.iterateData = function (loopData) {
 89     var tempText = theCall.htmlStructure[0];
 90     for (var key in loopData) {
 91       if (loopData.hasOwnProperty(key)) {
 92         var val = loopData[key];
 93         var keyText = this.addClassToConstruct(theCall.htmlStructure[2],key)+key+theCall.htmlStructure[4];
 94         if ((typeof(val)==='object') && val !== null) {
 95           tempText += (keyText + this.iterateData(val));
 96         } else {
 97           tempText += (keyText + val);
 98         }
 99         tempText += theCall.htmlStructure[3];
100       }
101     }
102     tempText += theCall.htmlStructure[1];
103     return tempText;
104   }
105   /**
106     * Sets the HTML construct based on the provided parameter (ul, ol, div, or dl).
107   */
108   this.setConstruct = function (htmlConstruct) {
109     var constructKey = '<'+htmlConstruct.toLowerCase()+'>';
110     var constructMap = [
111       ['<ul>','</ul>','<li>','</li>',' : '],
112       ['<ol>','</ol>','<li>','</li>',' : '],
113       ['<div>','</div>','<div>','</div>',''],
114       ['<dl>','</dl>','<dt>','</dd>','</dt><dd>']
115     ];
116     for (var key in constructMap) {
117       if (constructMap.hasOwnProperty(key)) {
118         var thisMap = constructMap[key];
119         if (thisMap[0] == constructKey) {
120           theCall.htmlStructure = thisMap;
121           break;
122         }
123       }
124     }
125   }
126   /**
127     * Adds a CSS class to the rendered object based on the element's name.
128   */
129   this.addClassToConstruct = function (rawTag,newClass) {
130     return rawTag.replace('>',' class="'+newClass+'">');
131   }
132 }

134 /**
135   * Handles and stores authentication information.
136 */
137 function ComApigeeAuthHandler() {
138   var theHandler = this;
139   this.userName = '';
140   this.userPass = '';
141   /**
142     * Checks to see if the browser supports localStorage; returns true or false accordingly.
143   */
144   this.checkLocalStorage = function () {
145     try {
146       return 'localStorage' in window && window.localStorage !== null;
147     } catch (e) {
148       return false;
149     }
150   }
151   this.doesLocalStorage = this.checkLocalStorage();
152   /**
153     * If the browser supports localStorage and the credentials exist in LS, confirms that the user wants to keep using these, and either sets userName and userPass accordingly or calls getCredentials to query for them.
154     * If the browser does not support localStorage (or the credentials do not exist), call getCredentials to query for them.
155   */
156   this.init = function () {
157     var doSet = true;
158     if(theHandler.doesLocalStorage) {
159       if(localStorage.userName && localStorage.userPass) {
160         theHandler.userName = localStorage.userName;
161         theHandler.userPass = localStorage.userPass;
162         doSet = !(confirm("Existing Apigee credentials (\""+theHandler.userName+"\") detected.\nUse these?"));
163       }
164     }
165     if(doSet) {
166       this.getCredentials();
167     }
168   }
169   /**
170     * Prompts for username and password.
171     * Passes prompted data into setCredentials.
172   */
173   this.getCredentials = function () {
174     theHandler.userName = prompt("Log into your Apigee Source account.\nEmail:",theHandler.userName);
175     theHandler.userPass = prompt("Password:",theHandler.userPass);
176     this.setCredentials();
177   }
178   /**
179     * If the browser supports localStorage, store the prompted data.
180   */
181   this.setCredentials = function () {
182     if(theHandler.doesLocalStorage) {
183       localStorage.userName = theHandler.userName;
184       localStorage.userPass = theHandler.userPass;
185     }
186   }
187   this.init();
188 }
